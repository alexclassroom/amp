public function __construct( $dom, $args = [] ) {
	$this->dom  = $dom;
	$this->args = array_merge( $this->DEFAULT_ARGS, $args );
	if ( ! empty( $this->args['use_document_element'] ) ) {
		$this->root_element = $this->dom->documentElement;
	} else {
		$this->root_element = $this->dom->body;
	}
}

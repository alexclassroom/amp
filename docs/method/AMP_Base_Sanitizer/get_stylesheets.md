```php
public function get_stylesheets() {
	$stylesheets = [];
	foreach ( $this->get_styles() as $selector => $properties ) {
		$stylesheet = sprintf( '%s { %s }', $selector, implode( '; ', $properties ) . ';' );
		$stylesheets[ md5( $stylesheet ) ] = $stylesheet;
	}
	return $stylesheets;
}
```

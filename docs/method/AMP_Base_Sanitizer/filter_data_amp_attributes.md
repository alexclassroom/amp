public function filter_data_amp_attributes( $attributes, $amp_data ) {
	if ( isset( $amp_data['layout'] ) ) {
		$attributes['data-amp-layout'] = $amp_data['layout'];
	}
	if ( isset( $amp_data['noloading'] ) ) {
		$attributes['data-amp-noloading'] = '';
	}
	return $attributes;
}

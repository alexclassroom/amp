```php
public static function get_node_attributes_as_assoc_array( $node ) {
	$attributes = [];
	if ( ! $node->hasAttributes() ) {
		return $attributes;
	}
	foreach ( $node->attributes as $attribute ) {
		$attributes[ $attribute->nodeName ] = $attribute->nodeValue;
	}
	return $attributes;
}
```

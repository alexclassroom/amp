public function has_path_suffix( $url ) {
	$path    = wp_parse_url( $url, PHP_URL_PATH );
	$pattern = sprintf(
		':/%s/?$:',
		preg_quote( amp_get_slug(), ':' )
	);
	return (bool) preg_match( $pattern, $path );
}

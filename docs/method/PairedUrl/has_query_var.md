```php
public function has_query_var( $url ) {
	$parsed_url = wp_parse_url( $url );
	if ( ! empty( $parsed_url['query'] ) ) {
		$query_vars = [];
		wp_parse_str( $parsed_url['query'], $query_vars );
		if ( isset( $query_vars[ amp_get_slug() ] ) ) {
			return true;
		}
	}
	return false;
}
```
